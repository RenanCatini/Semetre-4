-- ########## Comandos de Definição de Dados (DDL) ##########

[cite_start]-- Cria um novo banco de dados chamado hostelapp_db [cite: 135, 138, 201]
create database hostelapp_db;

[cite_start]-- Cria uma nova tabela chamada GUEST com suas respectivas colunas [cite: 219, 220, 221, 222, 223, 224, 225, 226]
CREATE TABLE GUEST(
    GUEST_ID BIGINT AUTO_INCREMENT PRIMARY KEY,
    FIRST_NAME VARCHAR(64),
    LAST_NAME VARCHAR(64),
    EMAIL_ADDRESS VARCHAR(64),
    ADDRESS VARCHAR(64),
    COUNTRY VARCHAR(32),
    STATE VARCHAR(12),
    PHONE_NUMBER VARCHAR(24)
);

[cite_start]-- Altera a senha do usuário 'root' (alternativa) [cite: 86]
ALTER USER 'root'@'localhost' IDENTIFIED BY '12345';

-- ########## Comandos de Controle de Dados (DCL) ##########

[cite_start]-- Cria um novo usuário chamado 'florentino' com a senha 'secret' [cite: 578, 664]
CREATE USER 'florentino'@'%' IDENTIFIED BY 'secret';

[cite_start]-- Concede todos os privilégios no banco de dados 'hostelapp_db' ao usuário 'florentino' de qualquer host [cite: 584, 664]
GRANT ALL PRIVILEGES ON hostelapp_db.* TO 'florentino'@'%';

[cite_start]-- Concede todos os privilégios no banco de dados 'hostelapp_db' ao usuário 'florentino' apenas do localhost [cite: 61, 664]
